  \subsection{Types of Functions}

    \begin{itemize}
      \item \textbf{One layer}: linear decision
      \item \textbf{A hidden layer}: any continuous function
      \item \textbf{More layers}: finer boundaries
      \item Flatter layers are harder to design
    \end{itemize}
